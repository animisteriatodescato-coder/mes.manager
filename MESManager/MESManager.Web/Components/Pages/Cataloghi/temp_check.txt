@code {
    private List<QuoteListDto> quotes = new();
    private QuoteListFilter filter = new() { PageSize = 100 };
    private DateRange? dateRange;
    private int totalCount = 0;
    private bool isLoading = false;
    private DateTime lastUpdate = DateTime.MinValue;
    
    // Import dialog
    private bool showImportDialog = false;
    private IBrowserFile? selectedFile;
    private string? importPriceListName;
    private string? importPriceListVersion;
    private ExcelImportResultDto? importResult;
    private bool isImporting = false;

    protected override async Task OnInitializedAsync()
    {
        PageToolbarService.SetActivePage("preventivi", this);
        await LoadData();
    }

    private async Task LoadData()
    {
        isLoading = true;
        StateHasChanged();
        
        try
        {
            // Costruisci query string
            var queryParams = new List<string>();
            
            if (!string.IsNullOrWhiteSpace(filter.SearchText))
                queryParams.Add($"searchText={Uri.EscapeDataString(filter.SearchText)}");
            
            if (filter.Status.HasValue)
                queryParams.Add($"status={(int)filter.Status.Value}");
            
            if (filter.DateFrom.HasValue)
                queryParams.Add($"dateFrom={filter.DateFrom.Value:yyyy-MM-dd}");
            
            if (filter.DateTo.HasValue)
                queryParams.Add($"dateTo={filter.DateTo.Value:yyyy-MM-dd}");
            
            queryParams.Add($"pageSize=500"); // Carica tutto per ora
            
            var url = "api/Quotes" + (queryParams.Any() ? "?" + string.Join("&", queryParams) : "");
            
            var result = await Http.GetFromJsonAsync<QuoteListResult>(url);
            
            if (result != null)
            {
                quotes = result.Items;
                totalCount = result.TotalCount;
            }
            
            lastUpdate = DateTime.Now;
        }
        catch (Exception ex)
        {
            Snackbar.Add($"Errore caricamento preventivi: {ex.Message}", Severity.Error);
        }
        finally
        {
            isLoading = false;
            StateHasChanged();
        }
    }

    private async Task OnSearchChanged(string value)
    {
        filter.SearchText = value;
        await LoadData();
    }

    private async Task OnFilterChanged(QuoteStatus? status)
    {
        filter.Status = status;
        await LoadData();
    }

    private async Task OnDateRangeChanged(DateRange? range)
    {
        dateRange = range;
        filter.DateFrom = range?.Start;
        filter.DateTo = range?.End;
        await LoadData();
    }

    private void NuovoPreventivo()
    {
        Nav.NavigateTo("/preventivi/nuovo");
    }

    private void OpenQuote(Guid id)
    {
        Nav.NavigateTo($"/preventivi/{id}");
    }

    private void EditQuote(Guid id)
    {
        Nav.NavigateTo($"/preventivi/{id}/edit");
    }

    private async Task DuplicateQuote(Guid id)
    {
        try
        {
            var response = await Http.PostAsync($"api/Quotes/{id}/duplicate", null);
            if (response.IsSuccessStatusCode)
            {
                var newQuote = await response.Content.ReadFromJsonAsync<QuoteDto>();
                Snackbar.Add($"Preventivo duplicato: {newQuote?.Number}", Severity.Success);
                await LoadData();
            }
            else
            {
                var error = await response.Content.ReadAsStringAsync();
                Snackbar.Add($"Errore duplicazione: {error}", Severity.Error);
            }
        }
        catch (Exception ex)
        {
            Snackbar.Add($"Errore: {ex.Message}", Severity.Error);
        }
    }

    private async Task DownloadPdf(Guid id, string number)
    {
        try
        {
            var response = await Http.GetAsync($"api/Quotes/{id}/pdf");
            if (response.IsSuccessStatusCode)
            {
                var content = await response.Content.ReadAsByteArrayAsync();
                var fileName = $"Preventivo_{number}.pdf";
                
                // Download tramite JS
                await JSRuntime.InvokeVoidAsync("downloadFile", content, fileName, "application/pdf");
                
                Snackbar.Add("PDF scaricato", Severity.Success);
            }
            else
            {
                Snackbar.Add("Errore generazione PDF", Severity.Error);
            }
        }
        catch (Exception ex)
        {
            Snackbar.Add($"Errore: {ex.Message}", Severity.Error);
        }
    }

    private async Task DeleteQuote(Guid id, string number)
    {
        // Conferma eliminazione
        var confirmed = await JSRuntime.InvokeAsync<bool>("confirm", $"Eliminare il preventivo {number}?");
        if (!confirmed) return;
        
        try
        {
            var response = await Http.DeleteAsync($"api/Quotes/{id}");
            if (response.IsSuccessStatusCode)
            {
                Snackbar.Add($"Preventivo {number} eliminato", Severity.Success);
                await LoadData();
            }
            else
            {
                Snackbar.Add("Errore eliminazione", Severity.Error);
            }
        }
        catch (Exception ex)
        {
            Snackbar.Add($"Errore: {ex.Message}", Severity.Error);
        }
    }

    private void OpenImportDialog()
    {
        selectedFile = null;
        importPriceListName = null;
        importPriceListVersion = null;
        importResult = null;
        showImportDialog = true;
    }

    private void OnFileSelected(IBrowserFile file)
    {
        selectedFile = file;
        importResult = null;
    }

    private async Task ImportPriceList()
    {
        if (selectedFile == null) return;
        
        isImporting = true;
        importResult = null;
        StateHasChanged();
        
        try
        {
            using var content = new MultipartFormDataContent();
            
            var fileContent = new StreamContent(selectedFile.OpenReadStream(maxAllowedSize: 50 * 1024 * 1024));
            fileContent.Headers.ContentType = new System.Net.Http.Headers.MediaTypeHeaderValue(
                selectedFile.ContentType ?? "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet");
            
            content.Add(fileContent, "file", selectedFile.Name);
            
            if (!string.IsNullOrWhiteSpace(importPriceListName))
                content.Add(new StringContent(importPriceListName), "name");
            
            if (!string.IsNullOrWhiteSpace(importPriceListVersion))
                content.Add(new StringContent(importPriceListVersion), "version");
            
            var response = await Http.PostAsync("api/PriceLists/import", content);
            
            importResult = await response.Content.ReadFromJsonAsync<ExcelImportResultDto>();
            
            if (importResult?.Success == true)
            {
                Snackbar.Add($"Listino '{importResult.PriceListName}' importato con successo!", Severity.Success);
            }
        }
        catch (Exception ex)
        {
            importResult = new ExcelImportResultDto 
            { 
                Success = false, 
                ErrorMessage = ex.Message 
            };
        }
        finally
        {
            isImporting = false;
            StateHasChanged();
        }
    }

    private Color GetStatusColor(QuoteStatus status)
    {
        return status switch
        {
            QuoteStatus.Draft => Color.Default,
            QuoteStatus.Sent => Color.Info,
            QuoteStatus.Accepted => Color.Success,
            QuoteStatus.Rejected => Color.Error,
            QuoteStatus.Expired => Color.Warning,
            _ => Color.Default
        };
    }

    public async ValueTask DisposeAsync()
    {
        PageToolbarService.SetActivePage("preventivi", null);
    }
}
